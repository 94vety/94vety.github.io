# JavaScript 中的 apply

```bash
# 定义数据函数
var obj = {
    value: 'obj'
}

function show (name, age) {
    console.log(this.value)

    return {
        name,
        age
    }
}

# 测试
show.apply(obj, ['xiuluo', 16])
# obj                                show 函数中 console.log 打印的值
# { name: 'xiuluo', age: 16 }        show 函数返回的值
```

***

# 自定义 myCall

```bash
Function.prototype.myApply(ctx, arr) {
    # 拿到myApply的第一个参数(需要绑定的对象),传入null则绑定window
    let ctx = ctx || window;
    ctx.fn = this;
    # {value: "obj", fn: ƒ}

    if (!arr) {
        const result = ctx.fn();
        delete ctx.fn;
        return result;
    }

    const args = [];
    for (let i = 0; i < arr.length; i++) {
        args.push('arr[' + i + ']')
    }
    # args 为 [ 'arr[1]', 'arr[2]'... ]

    let result = eval('ctx.fn(' + args.join(',') + ')')
    delete ctx.fn;
    return result;
}
```

***

# 使用自定义的 myCall 测试

```bash
show.myApply(obj, 'xiuluo', 16)
# obj                                show 函数中 console.log 打印的值
# { name: 'xiuluo', age: 16 }        show 函数返回的值
```
